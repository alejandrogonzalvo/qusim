def evaluate( config: FomConfig, columns: dict[str, numpy.ndarray]) -> numpy.ndarray: 
140def evaluate(config: FomConfig, columns: dict[str, np.ndarray]) -> np.ndarray:
141    """Vectorised FoM evaluation over a column-wise primitive table.
142
143    ``columns[k]`` must be a 1-D ndarray of the same length for every ``k``.
144    Returns a float64 array of the same length; division-by-zero becomes NaN.
145    """
146    if not columns:
147        raise FomError("no primitives available")
148    size = next(iter(columns.values())).shape[0]
149    allowed: set[str] = set(columns.keys())
150    env: dict[str, Any] = dict(columns)
151
152    for raw_name, raw_expr in config.intermediates:
153        name = (raw_name or "").strip()
154        if not name:
155            continue
156        if not name.isidentifier():
157            raise FomError(f"invalid intermediate name: {name!r}")
158        if name in allowed:
159            raise FomError(f"intermediate {name!r} shadows a primitive")
160        code = _compile_expr(raw_expr, allowed)
161        value = _eval_compiled(code, env)
162        arr = np.asarray(value, dtype=float)
163        if arr.ndim == 0:
164            arr = np.full(size, float(arr))
165        elif arr.shape != (size,):
166            raise FomError(f"intermediate {name!r} produced wrong shape {arr.shape}")
167        env[name] = arr
168        allowed.add(name)
169
170    num_code = _compile_expr(config.numerator or "1", allowed)
171    den_code = _compile_expr(config.denominator or "1", allowed)
172    num = np.asarray(_eval_compiled(num_code, env), dtype=float)
173    den = np.asarray(_eval_compiled(den_code, env), dtype=float)
174    if num.ndim == 0:
175        num = np.full(size, float(num))
176    if den.ndim == 0:
177        den = np.full(size, float(den))
178    with np.errstate(divide="ignore", invalid="ignore"):
179        out = num / den
180    return np.where(np.isfinite(out), out, np.nan)

268    """Evaluate the FoM at every sweep point and return a ``FomResult``.
269
270    Never raises — user-facing errors are surfaced via ``FomResult.error``.
271    """
272    from quadris.dse.flatten import flatten_sweep_to_table as _flatten_sweep_to_table
273
274    try:
275        metric_keys, outputs, rows = _flatten_sweep_to_table(sweep_data)
276    except Exception as exc:  # pragma: no cover - defensive
277        return FomResult(values=None, error=f"could not flatten sweep: {exc}")
278
279    primitives = list(metric_keys) + list(outputs)
280    if rows.size == 0:
281        return FomResult(values=None, primitives=primitives, error="sweep is empty")
282
283    columns: dict[str, np.ndarray] = {}
284    for j, k in enumerate(metric_keys):
285        columns[k] = rows[:, j]
286    for j, k in enumerate(outputs):
287        columns[k] = rows[:, len(metric_keys) + j]
288
289    try:
290        values = evaluate(config, columns)
291    except FomError as exc:
292        return FomResult(values=None, primitives=primitives, error=str(exc))
293    except Exception as exc:  # arithmetic issues, etc.
294        return FomResult(values=None, primitives=primitives, error=f"evaluation error: {exc}")
295
296    return FomResult(values=values, primitives=primitives, error=None)

def pareto_front_mask(num: numpy.ndarray, den: numpy.ndarray) -> numpy.ndarray: 
19def pareto_front_mask(num: np.ndarray, den: np.ndarray) -> np.ndarray:
20    """Return a boolean mask selecting Pareto-optimal points under
21    *maximize* ``num`` and *minimize* ``den``.
22
23    O(N²) — fine for the ≤4096-point sweeps the engine produces. For
24    ten-thousand-plus sweeps swap in a dimension-sweep algorithm.
25
26    A point ``i`` is dominated iff there exists ``j`` with
27    ``num[j] >= num[i]`` and ``den[j] <= den[i]`` and at least one
28    inequality is strict.
29    """
30    num = np.asarray(num, dtype=float)
31    den = np.asarray(den, dtype=float)
32    n = num.shape[0]
33    if n == 0:
34        return np.zeros(0, dtype=bool)
35    num_arr = num.reshape(-1, 1)
36    den_arr = den.reshape(-1, 1)
37    ge_num = num_arr.T >= num_arr
38    le_den = den_arr.T <= den_arr
39    strict = (num_arr.T > num_arr) | (den_arr.T < den_arr)
40    dominated = (ge_num & le_den & strict).any(axis=1)
41    return ~dominated

 44def pareto_front(
 45    sweep: Any,
 46    objective_x: str,
 47    objective_y: str,
 48) -> dict[str, np.ndarray]:
 49    """Compute the Pareto front of a sweep over two output metrics.
 50
 51    Parameters
 52    ----------
 53    sweep
 54        A :class:`quadris.dse.SweepResult` (which knows how to convert
 55        itself via ``.to_sweep_data()``), or any sweep dict that
 56        :func:`flatten_sweep_to_table` accepts.
 57    objective_x, objective_y
 58        Output metric keys (e.g. ``"total_epr_pairs"``,
 59        ``"overall_fidelity"``). Each axis is min/max-imised according to
 60        :data:`quadris.dse.axes.PARETO_METRIC_ORIENTATION`.
 61
 62    Returns
 63    -------
 64    dict
 65        ``{"x": ..., "y": ..., "mask": ..., "axes": {<axis_key>: ...}}``
 66        where ``mask`` is the Pareto-optimal selector over the flattened
 67        sweep, and ``axes`` carries the swept-axis values for each point
 68        (handy for hover labels in custom plots).
 69    """
 70    if hasattr(sweep, "to_sweep_data"):
 71        sweep_dict = sweep.to_sweep_data()
 72    elif hasattr(sweep, "as_dict"):
 73        sweep_dict = sweep.as_dict()
 74    else:
 75        sweep_dict = sweep
 76
 77    metric_keys, outputs, rows = flatten_sweep_to_table(sweep_dict)
 78    if rows.size == 0:
 79        return {"x": np.empty(0), "y": np.empty(0), "mask": np.empty(0, dtype=bool), "axes": {}}
 80
 81    if objective_x not in outputs or objective_y not in outputs:
 82        raise ValueError(
 83            f"output metric not in sweep — got {objective_x!r}/{objective_y!r}, "
 84            f"sweep has {outputs}"
 85        )
 86
 87    x = rows[:, len(metric_keys) + outputs.index(objective_x)]
 88    y = rows[:, len(metric_keys) + outputs.index(objective_y)]
 89
 90    # Convert to (max num, min den) form expected by pareto_front_mask.
 91    x_dir = PARETO_METRIC_ORIENTATION.get(objective_x, "max")
 92    y_dir = PARETO_METRIC_ORIENTATION.get(objective_y, "max")
 93    num = y if y_dir == "max" else -y
 94    den = x if x_dir == "min" else -x
 95
 96    mask = pareto_front_mask(num, den)
 97
 98    axes_cols: dict[str, np.ndarray] = {
 99        k: rows[:, j] for j, k in enumerate(metric_keys)
100    }
101
102    return {"x": x, "y": y, "mask": mask, "axes": axes_cols}
