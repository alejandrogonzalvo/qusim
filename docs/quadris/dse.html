quadris.dse

Design Space Exploration (DSE) toolkit.

Build and run multi-dimensional sweeps over circuit, topology, and noise parameters; cache the expensive structural pass; cheaply re-evaluate noise configs from cached mappings.

Logical-first parameterization

The user pins exactly one of num_cores or qubits_per_core; the unpinned axis is deduced so the chip fits num_logical_qubits even as the user sweeps comm/buffer overhead. num_qubits (= num_cores · qubits_per_core) and idle_reserved_qubits are derived outputs.

Typical usage::

from quadris.dse import DSEEngine

engine = DSEEngine()
sweep = engine.sweep_nd(
    cached=None,
    sweep_axes=[
        ("num_cores",      1, 8),     # cold: pinned cores
        ("two_gate_error", -4, -2),   # hot: log10 endpoints
    ],
    fixed_noise={},
    cold_config={
        "circuit_type": "qft",
        "num_logical_qubits": 32,
        "qubits_per_core": 16,
        "num_cores": 4,                # overridden by the sweep
        "pin_axis": "cores",
        "topology_type": "ring",
        "intracore_topology": "all_to_all",
        "placement_policy": "spectral",
        "routing_algorithm": "hqa_sabre",
        "seed": 0,
    },
)
print(sweep.shape, sweep.metric_keys)

130    """Wraps quadris's mapper+estimator pipeline for Design Space Exploration.
131
132    Cold path — build circuit + topology, route via the selected backend,
133                cache structural results.
134    Hot path  — call ``estimate_hardware_fidelity`` directly with cached
135                data and a new noise dict.

137    Sweeps  — :meth:`sweep_nd` is the canonical entry point. The 1/2/3-D
138              variants are kept as legacy wrappers around it.

155    def run_cold(
156        self,
157        circuit_type: str,
158        num_logical_qubits: int,
159        num_cores: int,
160        qubits_per_core: int,
161        topology_type: str,
162        placement_policy: str,
163        seed: int,
164        noise: Optional[dict] = None,
165        intracore_topology: str = "all_to_all",
166        routing_algorithm: str = "hqa_sabre",
167        communication_qubits: int = 1,
168        buffer_qubits: int = 1,
169        pin_axis: str = DEFAULT_PIN_AXIS,
170        custom_qasm: Optional[str] = None,
171    ) -> CachedMapping:
172        """Build circuit + topology, route, cache. Returns a CachedMapping
173        that can be reused for many hot-path fidelity evaluations.
174
175        Logical-first parameterization: ``num_logical_qubits`` is the
176        algorithm size and is held constant during a sweep. Either
177        ``num_cores`` or ``qubits_per_core`` is pinned (per ``pin_axis``);
178        the unpinned axis is deduced via :func:`_resolve_architecture`
179        so the device fits the circuit. ``num_qubits`` (= ``num_cores ·
180        qubits_per_core``) is a derived field, written into the cold
181        config below.
182
183        Routing dispatches via :mod:`quadris.dse.backends` — pass
184        ``routing_algorithm="telesabre"`` to use the C-library mapper
185        instead of HQA+SABRE.
186
187        Raises ``ValueError`` when the configuration is infeasible
188        (e.g. B > K, or no ``nc`` satisfies the fixpoint with qpc
189        pinned). Sweep cells catch this and write a NaN row so the
190        whole sweep continues.
191        """
192        # Custom circuit overrides logical_qubits from the QASM payload.
193        if circuit_type == "custom":
194            if not custom_qasm:
195                raise ValueError("circuit_type='custom' requires a custom_qasm string")
196            from qiskit import qasm2
197            _custom_circ = qasm2.loads(custom_qasm)
198            num_logical_qubits = int(_custom_circ.num_qubits)
199
200        # Build a working config and run the deduction layer to fill in
201        # the unpinned axis + num_qubits.
202        cold_cfg = {
203            "circuit_type": circuit_type,
204            "num_logical_qubits": int(num_logical_qubits),
205            "num_cores": int(num_cores),
206            "qubits_per_core": int(qubits_per_core),
207            "topology_type": topology_type,
208            "intracore_topology": intracore_topology,
209            "placement_policy": placement_policy,
210            "seed": int(seed),
211            "routing_algorithm": routing_algorithm,
212            "communication_qubits": int(communication_qubits),
213            "buffer_qubits": int(buffer_qubits),
214            "pin_axis": pin_axis or DEFAULT_PIN_AXIS,
215            "custom_qasm": custom_qasm,
216        }
217        feasibility = _resolve_architecture(cold_cfg)
218        if not feasibility["feasible"]:
219            raise ValueError(
220                f"Infeasible architecture: {feasibility['reason']}"
221            )
222
223        # Cache key built from the resolved cfg so changing the *pinned*
224        # axis vs the *derived* axis produces the same key when they
225        # land on the same device.
226        qasm_fp = (
227            hashlib.sha256(custom_qasm.encode("utf-8")).hexdigest()[:16]
228            if custom_qasm else None
229        )
230        config_key = (
231            circuit_type,
232            cold_cfg["num_qubits"],
233            cold_cfg["num_cores"],
234            cold_cfg["qubits_per_core"],
235            topology_type,
236            intracore_topology,
237            placement_policy,
238            int(seed),
239            routing_algorithm,
240            cold_cfg["communication_qubits"],
241            cold_cfg["buffer_qubits"],
242            cold_cfg["num_logical_qubits"],
243            qasm_fp,
244        )
245        if self._cache is not None and self._cache.config_key == config_key:
246            return self._cache
247
248        cached = _compile_one(cold_cfg, noise, config_key)
249        # Stamp the resolved architectural metrics on the cache so
250        # ``run_hot`` (called directly or via the sweep grid) can ride
251        # them through into every result dict.
252        cached.num_qubits = cold_cfg["num_qubits"]
253        cached.derived_num_cores = cold_cfg["num_cores"]
254        cached.derived_qubits_per_core = cold_cfg["qubits_per_core"]
255        cached.idle_reserved_qubits = cold_cfg.get("idle_reserved_qubits", 0)
256        self._cache = cached
257        return self._cache

261    def run_hot(self, cached: CachedMapping, noise: dict) -> dict:
262        """Fast fidelity estimation reusing cached placements + distance matrix."""
263        merged = _merge_noise(noise)
264        gate_error_arr, gate_time_arr = _make_gate_arrays(cached.gate_names, merged)
265        result = quadris.estimate_fidelity_from_cache(
266            gs_sparse=cached.gs_sparse,
267            placements=cached.placements,
268            distance_matrix=cached.distance_matrix,
269            sparse_swaps=cached.sparse_swaps,
270            gate_error_arr=gate_error_arr,
271            gate_time_arr=gate_time_arr,
272            single_gate_error=merged["single_gate_error"],
273            two_gate_error=merged["two_gate_error"],
274            teleportation_error_per_hop=merged["teleportation_error_per_hop"],
275            single_gate_time=merged["single_gate_time"],
276            two_gate_time=merged["two_gate_time"],
277            teleportation_time_per_hop=merged["teleportation_time_per_hop"],
278            epr_error_per_hop=merged["epr_error_per_hop"],
279            measurement_error=merged["measurement_error"],
280            t1=merged["t1"],
281            t2=merged["t2"],
282            dynamic_decoupling=merged.get("dynamic_decoupling", False),
283            readout_mitigation_factor=merged["readout_mitigation_factor"],
284            classical_link_width=int(merged["classical_link_width"]),
285            classical_clock_freq_hz=merged["classical_clock_freq_hz"],
286            classical_routing_cycles=int(merged["classical_routing_cycles"]),
287        )
288        result["total_epr_pairs"] = cached.total_epr_pairs
289        result["total_swaps"] = cached.total_swaps
290        result["total_teleportations"] = cached.total_teleportations
291        result["total_network_distance"] = cached.total_network_distance
292        # Derived architectural metrics ride along on the cached mapping
293        # so each hot-path call can stamp them onto the result dict
294        # without re-deriving from cfg.
295        for k in ("num_qubits", "derived_num_cores",
296                  "derived_qubits_per_core", "idle_reserved_qubits"):
297            v = getattr(cached, k, None)
298            if v is not None:
299                result[k] = v
300        return result

302    def run_hot_batch(
303        self,
304        cached: CachedMapping,
305        noise_dicts: list[dict],
306        keep_grids: bool = False,
307    ) -> list[dict]:
308        """Batch fidelity estimation: chunked Rust calls for many noise configs.
309
310        Structural data is parsed once per chunk; chunks bound peak
311        memory instead of allocating all results at once. With
312        ``keep_grids=True`` the per-qubit grids are retained — the Rust
313        batch entry point is scalar-only, so this branch falls back to a
314        per-cell ``run_hot`` loop.
315        """
316        if keep_grids:
317            return [self.run_hot(cached, n) for n in noise_dicts]
318
319        gate_error_arr, gate_time_arr = _make_gate_arrays(
320            cached.gate_names, _merge_noise(noise_dicts[0]),
321        )
322        all_results: list[dict] = []
323
324        for start in range(0, len(noise_dicts), self._HOT_BATCH_CHUNK):
325            chunk = noise_dicts[start : start + self._HOT_BATCH_CHUNK]
326            merged_chunk = [_merge_noise(n) for n in chunk]
327
328            chunk_results = quadris.estimate_fidelity_from_cache_batch(
329                gs_sparse=cached.gs_sparse,
330                placements=cached.placements,
331                distance_matrix=cached.distance_matrix,
332                sparse_swaps=cached.sparse_swaps,
333                gate_error_arr=gate_error_arr,
334                gate_time_arr=gate_time_arr,
335                noise_dicts=merged_chunk,
336            )
337            for r in chunk_results:
338                r["total_epr_pairs"] = cached.total_epr_pairs
339                r["total_swaps"] = cached.total_swaps
340                r["total_teleportations"] = cached.total_teleportations
341                r["total_network_distance"] = cached.total_network_distance
342                # Stamp resolved architectural metrics so each cell in
343                # the sweep grid carries (cores, qpc, num_qubits, idle)
344                # alongside the fidelity numbers.
345                for k in ("num_qubits", "derived_num_cores",
346                          "derived_qubits_per_core", "idle_reserved_qubits"):
347                    v = getattr(cached, k, None)
348                    if v is not None:
349                        r[k] = v
350            all_results.extend(_strip_for_grid(r) for r in chunk_results)
351
352        return all_results

356    def _memory_capped_max_hot(self, max_hot: int | None) -> tuple[int, int]:
357        """Clamp ``max_hot`` to what current RAM can hold.
358
359        Returns ``(effective, requested)`` so callers can report when
360        the cap kicked in. ``requested`` mirrors the user's configured
361        value (or the registry default when ``None``); ``effective`` is
362        what the sweep should actually respect.
363
364        Calls ``_max_hot_points_for_memory`` via this module's binding
365        so test monkeypatches against ``quadris.dse.engine`` apply here.
366        """
367        requested = max_hot if max_hot is not None else MAX_TOTAL_POINTS_HOT
368        return min(requested, _max_hot_points_for_memory()), requested

370    def _metric_values(self, metric_key: str, low, high, n: int) -> np.ndarray:
371        if metric_key in CAT_METRIC_BY_KEY:
372            return np.array(low, dtype=object)
373        m = METRIC_BY_KEY[metric_key]
374        if m.log_scale:
375            return np.logspace(low, high, n)
376        vals = np.linspace(low, high, n)
377        if metric_key in self.INTEGER_KEYS:
378            vals = np.unique(np.round(vals).astype(int))
379        return vals

389    def _points_per_axis(self, ndim: int, has_cold: bool) -> int:
390        budget = MAX_TOTAL_POINTS_COLD if has_cold else MAX_TOTAL_POINTS_HOT
391        if ndim <= 3:
392            return self._sweep_points(ndim, has_cold)
393        n = max(MIN_POINTS_PER_AXIS, int(budget ** (1.0 / ndim)))
394        while n ** ndim > budget and n > MIN_POINTS_PER_AXIS:
395            n -= 1
396        return n

398    def _compute_axis_counts(
399        self, sweep_axes: list, has_cold: bool,
400        max_cold: int | None = None,
401        max_hot: int | None = None,
402    ) -> list[int]:
403        """Compute per-axis point counts using a split budget model.
404
405        Categorical axes have a fixed count (len of selected values).
406        Numeric cold-path axes are capped by ``max_cold``. Hot-path axes
407        share the remaining hot budget.
408        """
409        ndim = len(sweep_axes)
410        cold_budget = max_cold if max_cold is not None else MAX_COLD_COMPILATIONS
411        hot_budget = max_hot if max_hot is not None else MAX_TOTAL_POINTS_HOT
412
413        cat_indices, cold_indices, hot_indices = [], [], []
414        for i, ax in enumerate(sweep_axes):
415            key = ax[0]
416            if key in CAT_METRIC_BY_KEY:
417                cat_indices.append(i)
418            elif key in self.COLD_PATH_KEYS:
419                cold_indices.append(i)
420            else:
421                hot_indices.append(i)
422
423        counts: list[int] = [0] * ndim
424        for i in cat_indices:
425            counts[i] = len(sweep_axes[i][1])
426
427        cat_product = 1
428        for i in cat_indices:
429            cat_product *= counts[i]
430
431        effective_cold_budget = max(1, cold_budget // max(1, cat_product))
432        if cold_indices:
433            n_cold = len(cold_indices)
434            per_cold = max(MIN_POINTS_PER_AXIS, int(effective_cold_budget ** (1.0 / n_cold)))
435            for i in cold_indices:
436                key, low, high = sweep_axes[i]
437                natural = int(high) - int(low) + 1
438                counts[i] = max(MIN_POINTS_PER_AXIS, min(natural, per_cold))
439            cold_product = 1
440            for i in cold_indices:
441                cold_product *= counts[i]
442            while cold_product > effective_cold_budget:
443                largest = max(cold_indices, key=lambda i: counts[i])
444                counts[largest] = max(MIN_POINTS_PER_AXIS, counts[largest] - 1)
445                cold_product = 1
446                for i in cold_indices:
447                    cold_product *= counts[i]
448                if all(counts[i] == MIN_POINTS_PER_AXIS for i in cold_indices):
449                    break
450
451        cold_product = cat_product
452        for i in cold_indices:
453            cold_product *= counts[i]
454
455        if hot_indices:
456            n_hot = len(hot_indices)
457            min_total = (MIN_POINTS_PER_AXIS ** n_hot) * cold_product
458            if min_total > hot_budget:
459                raise RuntimeError(
460                    f"Hot budget too tight for this sweep: "
461                    f"{n_hot} hot axes at {MIN_POINTS_PER_AXIS} points each × "
462                    f"{cold_product} cold combinations = {min_total:,} points, "
463                    f"but max_hot={hot_budget:,}. "
464                    f"Raise Max hot evaluations to at least {min_total:,}, "
465                    f"reduce dimensions, or lower Max cold compilations."
466                )
467            effective_hot_budget = max(1, hot_budget // max(1, cold_product))
468            per_hot = max(MIN_POINTS_PER_AXIS, int(effective_hot_budget ** (1.0 / n_hot)))
469            while per_hot ** n_hot * cold_product > hot_budget and per_hot > MIN_POINTS_PER_AXIS:
470                per_hot -= 1
471            for i in hot_indices:
472                counts[i] = per_hot
473        elif not cold_indices and not cat_indices:
474            n = max(MIN_POINTS_PER_AXIS, int(hot_budget ** (1.0 / ndim)))
475            counts = [n] * ndim
476
477        return counts

481    def _eval_point(
482        self,
483        cold_config: dict,
484        noise: dict,
485        swept: dict[str, float],
486    ) -> dict:
487        """Evaluate one design point, re-running cold path if needed.
488
489        Returns the full result dict (per-qubit grids included so callers
490        that need them can capture before stripping). Returns a NaN row
491        when the resolved architecture is infeasible.
492        """
493        cfg = dict(cold_config)
494        hot_noise = dict(noise)
495        for k, v in swept.items():
496            if k in self.COLD_PATH_KEYS:
497                cfg[k] = int(v) if k in self.INTEGER_KEYS else v
498            else:
499                hot_noise[k] = v
500        feasibility = _resolve_architecture(cfg)
501        if not feasibility["feasible"]:
502            return _nan_result_row(reason=feasibility["reason"])
503        # Strip keys run_cold doesn't accept (the resolver writes some
504        # bookkeeping fields onto cfg).
505        accepted = {
506            "circuit_type", "num_logical_qubits", "num_cores", "qubits_per_core",
507            "topology_type", "intracore_topology", "placement_policy", "seed",
508            "routing_algorithm", "communication_qubits", "buffer_qubits",
509            "pin_axis", "custom_qasm",
510        }
511        run_kwargs = {k: cfg[k] for k in accepted if k in cfg}
512        try:
513            cached = self.run_cold(**run_kwargs, noise=hot_noise)

552        max_hot: int | None = None,

555        """N-dimensional sweep over arbitrary axes.
556
557        ``sweep_axes`` entries are either ``(metric_key, low, high)`` for
558        numeric axes, or ``(metric_key, values_list)`` for categorical axes.
559

561        with per-cell per-qubit fidelity grids + placements + effective
562        cold config so the topology view can colour nodes / replot the
563        device for any cell without re-running the engine.  Adds memory
564        cost roughly proportional to ``num_cells × num_layers × num_qubits``.
565        """
566        ndim = len(sweep_axes)
567        keys = [ax[0] for ax in sweep_axes]
568        has_cold = self._has_cold(*keys)
569
570        # Clamp user-requested max_hot to what RAM can hold before
571        # computing per-axis counts. Without this, an oversized budget
572        # silently produces a grid the kernel OOM-kills mid-sweep.
573        effective_max_hot, requested_max_hot = self._memory_capped_max_hot(max_hot)
574        try:
575            axis_counts = self._compute_axis_counts(
576                sweep_axes, has_cold, max_cold, effective_max_hot,
577            )
578        except RuntimeError as e:
579            if effective_max_hot < requested_max_hot:
580                raise RuntimeError(
581                    f"{e} Memory cap: hot budget reduced from requested "
582                    f"{requested_max_hot:,} to {effective_max_hot:,} to fit in "
583                    f"available RAM."
584                ) from e
585            raise
586

588        # Categorical axes: ax = (key, values_list) — 2-tuple.
589        # Numeric axes:     ax = (key, low, high)   — 3-tuple.
590        axes = []
591        for i, ax in enumerate(sweep_axes):
592            k = ax[0]
593            if k in CAT_METRIC_BY_KEY:
594                axes.append(self._metric_values(k, ax[1], None, axis_counts[i]))
595            else:
596                axes.append(self._metric_values(k, ax[1], ax[2], axis_counts[i]))

598        total = int(np.prod(shape))
599
600        grid = np.empty(shape, dtype=_RESULT_DTYPE)
601
602        def _make_swept(idx):
603            return {keys[d]: axes[d][idx[d]] for d in range(ndim)}
604
605        def _progress_params(swept):
606            out = {}
607            for k, v in swept.items():
608                try:
609                    out[k] = float(v)
610                except (TypeError, ValueError):
611                    out[k] = 0.0
612            return out
613

615        # topology view does an O(1) lookup instead of paying the cold
616        # compile a second time per slider move.
617        per_qubit_cells: dict[tuple, dict] = {}
618
619        def _capture(idx: tuple, full_res: dict, cell_swept: dict) -> None:
620            if not keep_per_qubit_grids:
621                return
622            cell_cfg = (
623                _resolve_cell_cold_cfg(cold_config, cell_swept)
624                if cold_config is not None else None
625            )
626            per_qubit_cells[tuple(idx)] = _extract_per_qubit(
627                full_res, cached=None, cold_cfg=cell_cfg,
628            )
629
630        if parallel and has_cold and cold_config is not None:
631            def _indexed_iter():
632                for idx in np.ndindex(shape):
633                    yield idx, _make_swept(idx)
634
635            rmap = self._parallel_cold_sweep(
636                cold_config, fixed_noise, _indexed_iter(), total,
637                progress_callback, max_workers,
638                keep_grids=keep_per_qubit_grids,
639            )
640            for idx in np.ndindex(shape):
641                full_res = rmap[idx]
642                grid[idx] = _result_to_row(full_res)
643                _capture(idx, full_res, _make_swept(idx))
644        elif has_cold and cold_config is not None:
645            count = 0
646            for idx in np.ndindex(shape):
647                swept = _make_swept(idx)
648                full_res = self._eval_point(cold_config, fixed_noise, swept)
649                grid[idx] = _result_to_row(full_res)
650                _capture(idx, full_res, swept)
651                count += 1
652                if progress_callback is not None:
653                    progress_callback(SweepProgress(
654                        completed=count,
655                        total=total,
656                        current_params=_progress_params(swept),
657                    ))
658        else:
659            # Pure hot-path: stream indices in chunks.
660            index_iter = iter(np.ndindex(shape))
661            completed = 0
662            while True:
663                chunk_indices = list(islice(index_iter, self._HOT_BATCH_CHUNK))
664                if not chunk_indices:
665                    break
666                noise_dicts = []
667                for idx in chunk_indices:
668                    noise = dict(fixed_noise)
669                    for d in range(ndim):
670                        noise[keys[d]] = axes[d][idx[d]]
671                    noise_dicts.append(noise)
672
673                chunk_results = self.run_hot_batch(
674                    cached, noise_dicts, keep_grids=keep_per_qubit_grids,
675                )
676                for i, idx in enumerate(chunk_indices):
677                    full_res = chunk_results[i]
678                    grid[idx] = _result_to_row(full_res)
679                    _capture(idx, full_res, _make_swept(idx))
680                    completed += 1
681                    if progress_callback is not None:
682                        progress_callback(SweepProgress(
683                            completed=completed,
684                            total=total,
685                            current_params=_progress_params(_make_swept(idx)),
686                        ))
687                del noise_dicts, chunk_results
688
689        # Always snapshot the engine inputs so the topology view's
690        # overlay can rebuild any cell on demand. ``cells`` (the per-
691        # cell per-qubit grids) is only attached when
692        # ``keep_per_qubit_grids=True``; without it, scrubbing costs a
693        # cold compile per cell, but the slider scaffolding still works
694        # — including after a session load where the heavy ``cells``
695        # dict is stripped before serialisation.
696        per_qubit_meta: dict | None = {
697            "cold_config": dict(cold_config) if cold_config else None,
698            "fixed_noise": dict(fixed_noise) if fixed_noise else None,
699            "axis_keys": list(keys),
700            "axis_values": [ax.tolist() for ax in axes],
701            "shape": list(shape),
702        }
703        if keep_per_qubit_grids:
704            per_qubit_meta["cells"] = per_qubit_cells
705
706        return SweepResult(
707            metric_keys=keys, axes=axes, grid=grid,
708            per_qubit_data=per_qubit_meta,
709        )

564        cost roughly proportional to ``num_cells × num_layers × num_qubits``.
565        """
566        ndim = len(sweep_axes)
567        keys = [ax[0] for ax in sweep_axes]
568        has_cold = self._has_cold(*keys)
569
570        # Clamp user-requested max_hot to what RAM can hold before
571        # computing per-axis counts. Without this, an oversized budget
572        # silently produces a grid the kernel OOM-kills mid-sweep.
573        effective_max_hot, requested_max_hot = self._memory_capped_max_hot(max_hot)
574        try:
575            axis_counts = self._compute_axis_counts(
576                sweep_axes, has_cold, max_cold, effective_max_hot,
577            )
578        except RuntimeError as e:
579            if effective_max_hot < requested_max_hot:
580                raise RuntimeError(
581                    f"{e} Memory cap: hot budget reduced from requested "
582                    f"{requested_max_hot:,} to {effective_max_hot:,} to fit in "
583                    f"available RAM."
584                ) from e
585            raise
586
587        # Per-axis value arrays.
588        # Categorical axes: ax = (key, values_list) — 2-tuple.
589        # Numeric axes:     ax = (key, low, high)   — 3-tuple.
590        axes = []
591        for i, ax in enumerate(sweep_axes):
592            k = ax[0]
593            if k in CAT_METRIC_BY_KEY:
594                axes.append(self._metric_values(k, ax[1], None, axis_counts[i]))
595            else:
596                axes.append(self._metric_values(k, ax[1], ax[2], axis_counts[i]))
597        shape = tuple(len(ax) for ax in axes)
598        total = int(np.prod(shape))
599
600        grid = np.empty(shape, dtype=_RESULT_DTYPE)
601
602        def _make_swept(idx):
603            return {keys[d]: axes[d][idx[d]] for d in range(ndim)}
604
605        def _progress_params(swept):
606            out = {}
607            for k, v in swept.items():
608                try:
609                    out[k] = float(v)
610                except (TypeError, ValueError):
611                    out[k] = 0.0
612            return out
613
614        # Per-cell per-qubit grids captured during the sweep so the
615        # topology view does an O(1) lookup instead of paying the cold
616        # compile a second time per slider move.
617        per_qubit_cells: dict[tuple, dict] = {}
618
619        def _capture(idx: tuple, full_res: dict, cell_swept: dict) -> None:
620            if not keep_per_qubit_grids:
621                return
622            cell_cfg = (
623                _resolve_cell_cold_cfg(cold_config, cell_swept)
624                if cold_config is not None else None
625            )
626            per_qubit_cells[tuple(idx)] = _extract_per_qubit(
627                full_res, cached=None, cold_cfg=cell_cfg,
628            )
629
630        if parallel and has_cold and cold_config is not None:
631            def _indexed_iter():
632                for idx in np.ndindex(shape):
633                    yield idx, _make_swept(idx)
634
635            rmap = self._parallel_cold_sweep(
636                cold_config, fixed_noise, _indexed_iter(), total,
637                progress_callback, max_workers,
638                keep_grids=keep_per_qubit_grids,
639            )
640            for idx in np.ndindex(shape):
641                full_res = rmap[idx]
642                grid[idx] = _result_to_row(full_res)
643                _capture(idx, full_res, _make_swept(idx))
644        elif has_cold and cold_config is not None:
645            count = 0
646            for idx in np.ndindex(shape):
647                swept = _make_swept(idx)
648                full_res = self._eval_point(cold_config, fixed_noise, swept)
649                grid[idx] = _result_to_row(full_res)
650                _capture(idx, full_res, swept)
651                count += 1
652                if progress_callback is not None:
653                    progress_callback(SweepProgress(
654                        completed=count,
655                        total=total,
656                        current_params=_progress_params(swept),
657                    ))
658        else:
659            # Pure hot-path: stream indices in chunks.
660            index_iter = iter(np.ndindex(shape))
661            completed = 0
662            while True:
663                chunk_indices = list(islice(index_iter, self._HOT_BATCH_CHUNK))
664                if not chunk_indices:
665                    break
666                noise_dicts = []
667                for idx in chunk_indices:
668                    noise = dict(fixed_noise)
669                    for d in range(ndim):
670                        noise[keys[d]] = axes[d][idx[d]]
671                    noise_dicts.append(noise)
672
673                chunk_results = self.run_hot_batch(
674                    cached, noise_dicts, keep_grids=keep_per_qubit_grids,
675                )
676                for i, idx in enumerate(chunk_indices):
677                    full_res = chunk_results[i]
678                    grid[idx] = _result_to_row(full_res)
679                    _capture(idx, full_res, _make_swept(idx))
680                    completed += 1
681                    if progress_callback is not None:
682                        progress_callback(SweepProgress(
683                            completed=completed,
684                            total=total,
685                            current_params=_progress_params(_make_swept(idx)),
686                        ))
687                del noise_dicts, chunk_results
688
689        # Always snapshot the engine inputs so the topology view's
690        # overlay can rebuild any cell on demand. ``cells`` (the per-
691        # cell per-qubit grids) is only attached when
692        # ``keep_per_qubit_grids=True``; without it, scrubbing costs a
693        # cold compile per cell, but the slider scaffolding still works
694        # — including after a session load where the heavy ``cells``
695        # dict is stripped before serialisation.
696        per_qubit_meta: dict | None = {
697            "cold_config": dict(cold_config) if cold_config else None,
698            "fixed_noise": dict(fixed_noise) if fixed_noise else None,
699            "axis_keys": list(keys),
700            "axis_values": [ax.tolist() for ax in axes],
701            "shape": list(shape),
702        }
703        if keep_per_qubit_grids:
704            per_qubit_meta["cells"] = per_qubit_cells
705
706        return SweepResult(
707            metric_keys=keys, axes=axes, grid=grid,
708            per_qubit_data=per_qubit_meta,
709        )

38def inter_core_neighbors(num_cores: int, inter_topology: str) -> list[list[int]]:
39    """Per-core list of unique neighbouring core indices."""
40    if num_cores < 2:
41        return [[] for _ in range(num_cores)]
42    nbrs: list[list[int]] = [[] for _ in range(num_cores)]
43    inter = (inter_topology or "ring").lower()
44    if inter == "ring":
45        for c in range(num_cores):
46            nbrs[c].append((c + 1) % num_cores)
47            if num_cores > 2:
48                nbrs[c].append((c - 1) % num_cores)
49    elif inter == "linear":
50        for c in range(num_cores):
51            if c + 1 < num_cores:
52                nbrs[c].append(c + 1)
53            if c - 1 >= 0:
54                nbrs[c].append(c - 1)
55    elif inter == "all_to_all":
56        for c in range(num_cores):
57            for c2 in range(num_cores):
58                if c2 != c:
59                    nbrs[c].append(c2)
60    elif inter == "grid":
61        side = math.ceil(math.sqrt(num_cores))
62        for c in range(num_cores):
63            row, col = divmod(c, side)
64            if col + 1 < side and c + 1 < num_cores:
65                nbrs[c].append(c + 1)
66            if col > 0 and c - 1 >= 0:
67                nbrs[c].append(c - 1)
68            if c + side < num_cores:
69                nbrs[c].append(c + side)
70            if c - side >= 0:
71                nbrs[c].append(c - side)
72    else:
73        for c in range(num_cores):
74            nbrs[c].append((c + 1) % num_cores)
75            if num_cores > 2:
76                nbrs[c].append((c - 1) % num_cores)
77    return [sorted(set(n)) for n in nbrs]
