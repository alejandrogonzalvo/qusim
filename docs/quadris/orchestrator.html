 51    def orchestrate(self, global_circuit: QuantumCircuit, hqa_mapping: List[List[int]]) -> tuple[QuantumCircuit, List[List[int]]]:
 52        """
 53        Executes the two-stage Global (HQA) to Local (SABRE) compilation pass.
 54        Returns None for the circuit (stitching disabled to prevent SABRE deadlocks), 
 55        and a sparse list of exactly when/where SABRE SWAPs occurred.
 56        """
 57        sub_circuits = self._partition_circuit(global_circuit, hqa_mapping)
 58        routed_sub_circuits = self._route_locals(sub_circuits)
 59        sparse_swaps = self._extract_sparse_swaps(global_circuit, routed_sub_circuits)
 60        return None, sparse_swaps

 66    def _build_core_padded_topologies(self) -> List[CouplingMap]:
 67        """
 68        Builds local core topologies from the global graph with star-graph padding 
 69        for foreign cores to act as parking space for inactive virtual qubits.
 70        """
 71        topologies = []
 72        # Total physical qubit count comes from the core_mapping (one entry
 73        # per physical qubit) rather than from edge endpoints — when the
 74        # architecture has no inter-core edges (e.g. K=0 in a small ring)
 75        # or sparse intra-core edges, the highest-indexed qubit can be
 76        # absent from the edge list and would underestimate ``nq``.
 77        nq = (max(self.core_mapping.keys()) + 1) if self.core_mapping else 0
 78
 79        for c in range(self.num_cores):
 80            topologies.append(self._build_single_core_topology(c, nq))
 81            
 82        return topologies
 83        
 84    def _build_single_core_topology(self, core_idx: int, nq: int) -> CouplingMap:
 85        cmap = CouplingMap()
 86        for i in range(nq):
 87            cmap.add_physical_qubit(i)
 88            
 89        core_nodes = [p for p, core in self.core_mapping.items() if core == core_idx]
 90        
 91        for u, v in self.full_coupling_map.get_edges():
 92            if self.core_mapping[u] == core_idx and self.core_mapping[v] == core_idx:
 93                cmap.add_edge(u, v)
 94
 95        boundary_nodes = [
 96            node for node in core_nodes
 97            if any(self.core_mapping[neighbor] != core_idx for neighbor in self.full_coupling_map.neighbors(node))
 98        ]
 99        
100        if not boundary_nodes:
101            boundary_nodes = core_nodes
102            
103        for i in range(nq):
104            if i in core_nodes:
105                continue
106            for b in boundary_nodes:
107                cmap.add_edge(i, b)
108                cmap.add_edge(b, i)
109                
110        return cmap

116    def _partition_circuit(self, global_circuit: QuantumCircuit, hqa_mapping: List[List[int]]) -> List[QuantumCircuit]:
117        """
118        Splits the global DAG into C local DAGs, injecting physical I/O primitives.
119        """
120        self._build_global_layout(global_circuit, hqa_mapping)
121        
122        dag = circuit_to_dag(global_circuit)
123        sub_dags = self._init_sub_dags(global_circuit)
124                
125        for t, layer in enumerate(dag.layers()):
126            t_prev = min(t - 1, len(hqa_mapping) - 1) if t > 0 else 0
127            t_curr = min(t, len(hqa_mapping) - 1)
128            
129            if t > 0:
130                self._inject_teleportation_if_moved(sub_dags, global_circuit, hqa_mapping[t_prev], hqa_mapping[t_curr], t)
131                
132            self._distribute_layer_ops(sub_dags, global_circuit, layer, hqa_mapping[t_curr], t)
133                    
134        return [dag_to_circuit(d) for d in sub_dags]

136    def _build_global_layout(self, global_circuit: QuantumCircuit, hqa_mapping: List[List[int]]):
137        self.global_layout_dict = {}
138        core_phys_available = {c: sorted([p for p, core in self.core_mapping.items() if core == c]) for c in range(self.num_cores)}
139        used_phys: set[int] = set()
140        all_phys = sorted(self.core_mapping.keys())
141        for q in range(global_circuit.num_qubits):
142            c = hqa_mapping[0][q]
143            if len(core_phys_available[c]) > 0:
144                p = core_phys_available[c].pop(0)
145            else:
146                # Core over-allocated by HQA — find any unused physical qubit
147                p = next((pp for pp in all_phys if pp not in used_phys), 0)
148            used_phys.add(p)
149            self.global_layout_dict[global_circuit.qubits[q]] = p

151    def _init_sub_dags(self, global_circuit: QuantumCircuit) -> List[DAGCircuit]:
152        sub_dags = [DAGCircuit() for _ in range(self.num_cores)]
153        for c in range(self.num_cores):
154            for qreg in global_circuit.qregs:
155                sub_dags[c].add_qreg(qreg)
156            for creg in global_circuit.cregs:
157                sub_dags[c].add_creg(creg)
158        return sub_dags

160    def _inject_teleportation_if_moved(self, sub_dags: List[DAGCircuit], global_circuit: QuantumCircuit, prev_mapping: List[int], current_mapping: List[int], t: int):
161        for q_idx in range(global_circuit.num_qubits):
162            src_core = prev_mapping[q_idx]
163            dst_core = current_mapping[q_idx]
164            
165            if src_core == dst_core:
166                continue
167                
168            qubit = global_circuit.qubits[q_idx]
169            self._inject_magnet_pull(sub_dags, global_circuit, prev_mapping, dst_core, src_core, qubit)
170            
171            sub_dags[src_core].apply_operation_back(TeledataSend(dst_core, t), [qubit])
172            sub_dags[dst_core].apply_operation_back(TeledataRecv(src_core, t), [qubit])
173
174    def _inject_magnet_pull(self, sub_dags: List[DAGCircuit], global_circuit: QuantumCircuit, prev_mapping: List[int], dst_core: int, src_core: int, qubit):
175        """
176        Injects a dummy two-qubit gate to bias SABRE toward routing the teleporting
177        qubit close to the core boundary before the teleportation occurs.
178
179        The trick: before emitting a TeledataSend, we insert a fake CX ('dummy_cx')
180        between the departing qubit and a qubit that already lives on the destination
181        core. SABRE sees this as a two-qubit dependency and tries to bring the two
182        operands adjacent — which effectively pulls the departing qubit toward the
183        boundary node, minimising the number of real SWAPs needed to reach the
184        teleportation point. The dummy_cx is stripped from the routed circuit by
185        _strip_dummy_cx before SWAP extraction.
186        """
187        magnets_in_dst = [i for i, c in enumerate(prev_mapping) if c == dst_core]
188        if not magnets_in_dst:
189            return
190
191        magnet_q = global_circuit.qubits[magnets_in_dst[0]]
192        dummy_cx = Instruction("dummy_cx", 2, 0, [])
193        sub_dags[src_core].apply_operation_back(dummy_cx, [qubit, magnet_q])
194        sub_dags[dst_core].apply_operation_back(dummy_cx, [qubit, magnet_q])
195
196    def _distribute_layer_ops(self, sub_dags: List[DAGCircuit], global_circuit: QuantumCircuit, layer, current_mapping: List[int], t: int):
197        for node in layer['graph'].op_nodes():
198            qargs = node.qargs
199            cores_involved = [current_mapping[global_circuit.find_bit(q).index] for q in qargs]
200            
201            if len(set(cores_involved)) == 1: 
202                self._apply_local_gate(sub_dags, node, cores_involved[0], t)
203            else: 
204                self._apply_intercore_gate(sub_dags, node, cores_involved, t)
205
206    def _apply_local_gate(self, sub_dags: List[DAGCircuit], node: DAGOpNode, core: int, t: int):
207        if hasattr(node.op, 'to_mutable'):
208            new_op = node.op.to_mutable()
209        else:
210            new_op = copy.copy(node.op)
211            
212        if hasattr(new_op, "label") and new_op.label:
213            new_op.label = f"{new_op.label}_L{t}"
214        else:
215            new_op.label = f"L{t}"
216        sub_dags[core].apply_operation_back(new_op, node.qargs, node.cargs)
217
218    def _apply_intercore_gate(self, sub_dags: List[DAGCircuit], node: DAGOpNode, cores_involved: List[int], t: int):
219        core_a, core_b = cores_involved[0], cores_involved[1]
220        q_a, q_b = node.qargs[0], node.qargs[1]
221        
222        sub_dags[core_a].apply_operation_back(TelegateHalf(core_b, node.op, True, t), [q_a])
223        sub_dags[core_b].apply_operation_back(TelegateHalf(core_a, node.op, False, t), [q_b])

229    def _route_locals(self, sub_circuits: List[QuantumCircuit]) -> List[QuantumCircuit]:
230        """
231        Executes Qiskit's SabreRouting pass heavily constrained by the physical topologies.
232        """
233        return [self._route_single_core(i, circ) for i, circ in enumerate(sub_circuits)]
234
235    def _route_single_core(self, core_idx: int, circ: QuantumCircuit) -> QuantumCircuit:
236        if not circ.data:
237            return circ
238
239        layout = Layout(self.global_layout_dict)
240        topo = self.core_topologies[core_idx]
241        pm = PassManager([
242            SetLayout(layout),
243            FullAncillaAllocation(topo),
244            EnlargeWithAncilla(),
245            ApplyLayout(),
246            SabreSwap(coupling_map=topo, heuristic='basic', seed=42),
247        ])
248
249        try:
250            import concurrent.futures
251            pool = concurrent.futures.ThreadPoolExecutor(max_workers=1)
252            future = pool.submit(pm.run, circ)
253            try:
254                routed = future.result(timeout=30)
255            except concurrent.futures.TimeoutError:
256                print(f"Warning: SABRE routing timed out on core {core_idx}, skipping local routing")
257                pool.shutdown(wait=False, cancel_futures=True)
258                return circ
259            finally:
260                pool.shutdown(wait=False)
261            return self._strip_dummy_cx(routed)
262        except Exception as e:
263            print(f"Warning: SABRE routing failed on core {core_idx}: {e}")
264            return circ
265            
266    def _strip_dummy_cx(self, routed: QuantumCircuit) -> QuantumCircuit:
267        cleaned_routed = QuantumCircuit(*routed.qregs, *routed.cregs)
268        for inst in routed.data:
269            if inst.operation.name != "dummy_cx":
270                cleaned_routed.append(inst)
271        return cleaned_routed
272
273    def _extract_sparse_swaps(self, global_circuit: QuantumCircuit, routed_sub_circuits: List[QuantumCircuit]) -> List[List[int]]:
274        sparse_swaps = []
275        for circ in routed_sub_circuits:
276            current_layer = 0
277            for inst in circ.data:
278                current_layer = self._update_current_layer(inst.operation, current_layer)
279                swap = self._get_sabre_swap_if_present(global_circuit, inst, current_layer)
280                if swap is not None:
281                    sparse_swaps.append(swap)
282        return sparse_swaps
283
284    def _update_current_layer(self, op: Instruction, current_layer: int) -> int:
285        if isinstance(op, (TeledataSend, TeledataRecv, TelegateHalf)):
286            return op.layer_idx
287            
288        if hasattr(op, 'label') and op.label and 'L' in op.label:
289            try:
290                return int(op.label.split('L')[-1])
291            except ValueError:
292                pass
293                
294        return current_layer
295
296    def _get_sabre_swap_if_present(self, global_circuit: QuantumCircuit, inst, current_layer: int) -> List[int] | None:
297        op = inst.operation
298        if op.name == 'swap' and not (hasattr(op, 'label') and op.label and 'L' in op.label):
299            # After ApplyLayout the circuit is physical: qubit indices are physical qubit indices.
300            # global_circuit.find_bit() cannot resolve them (different register), so read the
301            # index directly from the Qubit object instead.
302            q1 = inst.qubits[0]._index if hasattr(inst.qubits[0], '_index') else inst.qubits[0].index
303            q2 = inst.qubits[1]._index if hasattr(inst.qubits[1], '_index') else inst.qubits[1].index
304            return [current_layer, q1, q2]
305        return None

164            dst_core = current_mapping[q_idx]
165            
166            if src_core == dst_core:
167                continue
168                
169            qubit = global_circuit.qubits[q_idx]
170            self._inject_magnet_pull(sub_dags, global_circuit, prev_mapping, dst_core, src_core, qubit)
171            
172            sub_dags[src_core].apply_operation_back(TeledataSend(dst_core, t), [qubit])
173            sub_dags[dst_core].apply_operation_back(TeledataRecv(src_core, t), [qubit])
174
175    def _inject_magnet_pull(self, sub_dags: List[DAGCircuit], global_circuit: QuantumCircuit, prev_mapping: List[int], dst_core: int, src_core: int, qubit):
176        """
177        Injects a dummy two-qubit gate to bias SABRE toward routing the teleporting
178        qubit close to the core boundary before the teleportation occurs.
179
180        The trick: before emitting a TeledataSend, we insert a fake CX ('dummy_cx')
181        between the departing qubit and a qubit that already lives on the destination
182        core. SABRE sees this as a two-qubit dependency and tries to bring the two
183        operands adjacent — which effectively pulls the departing qubit toward the
184        boundary node, minimising the number of real SWAPs needed to reach the
185        teleportation point. The dummy_cx is stripped from the routed circuit by
186        _strip_dummy_cx before SWAP extraction.
187        """
188        magnets_in_dst = [i for i, c in enumerate(prev_mapping) if c == dst_core]
189        if not magnets_in_dst:
190            return
191
192        magnet_q = global_circuit.qubits[magnets_in_dst[0]]
193        dummy_cx = Instruction("dummy_cx", 2, 0, [])
194        sub_dags[src_core].apply_operation_back(dummy_cx, [qubit, magnet_q])
195        sub_dags[dst_core].apply_operation_back(dummy_cx, [qubit, magnet_q])
196
197    def _distribute_layer_ops(self, sub_dags: List[DAGCircuit], global_circuit: QuantumCircuit, layer, current_mapping: List[int], t: int):
198        for node in layer['graph'].op_nodes():
199            qargs = node.qargs
200            cores_involved = [current_mapping[global_circuit.find_bit(q).index] for q in qargs]
201            
202            if len(set(cores_involved)) == 1: 
203                self._apply_local_gate(sub_dags, node, cores_involved[0], t)
204            else: 
205                self._apply_intercore_gate(sub_dags, node, cores_involved, t)
206
207    def _apply_local_gate(self, sub_dags: List[DAGCircuit], node: DAGOpNode, core: int, t: int):
208        if hasattr(node.op, 'to_mutable'):
209            new_op = node.op.to_mutable()
210        else:
211            new_op = copy.copy(node.op)
212            
213        if hasattr(new_op, "label") and new_op.label:
214            new_op.label = f"{new_op.label}_L{t}"
215        else:
216            new_op.label = f"L{t}"
217        sub_dags[core].apply_operation_back(new_op, node.qargs, node.cargs)
218
219    def _apply_intercore_gate(self, sub_dags: List[DAGCircuit], node: DAGOpNode, cores_involved: List[int], t: int):
220        core_a, core_b = cores_involved[0], cores_involved[1]
221        q_a, q_b = node.qargs[0], node.qargs[1]
222        
223        sub_dags[core_a].apply_operation_back(TelegateHalf(core_b, node.op, True, t), [q_a])
224        sub_dags[core_b].apply_operation_back(TelegateHalf(core_a, node.op, False, t), [q_b])

230    def _route_locals(self, sub_circuits: List[QuantumCircuit]) -> List[QuantumCircuit]:
231        """
232        Executes Qiskit's SabreRouting pass heavily constrained by the physical topologies.
233        """
234        return [self._route_single_core(i, circ) for i, circ in enumerate(sub_circuits)]
235
236    def _route_single_core(self, core_idx: int, circ: QuantumCircuit) -> QuantumCircuit:
237        if not circ.data:
238            return circ
239
240        layout = Layout(self.global_layout_dict)
241        topo = self.core_topologies[core_idx]
242        pm = PassManager([
243            SetLayout(layout),
244            FullAncillaAllocation(topo),
245            EnlargeWithAncilla(),
246            ApplyLayout(),
247            SabreSwap(coupling_map=topo, heuristic='basic', seed=42),
248        ])
249
250        try:
251            import concurrent.futures
252            pool = concurrent.futures.ThreadPoolExecutor(max_workers=1)
253            future = pool.submit(pm.run, circ)
254            try:
255                routed = future.result(timeout=30)
256            except concurrent.futures.TimeoutError:
257                print(f"Warning: SABRE routing timed out on core {core_idx}, skipping local routing")
258                pool.shutdown(wait=False, cancel_futures=True)
259                return circ
260            finally:
261                pool.shutdown(wait=False)
262            return self._strip_dummy_cx(routed)
263        except Exception as e:
264            print(f"Warning: SABRE routing failed on core {core_idx}: {e}")
265            return circ
266            
267    def _strip_dummy_cx(self, routed: QuantumCircuit) -> QuantumCircuit:
268        cleaned_routed = QuantumCircuit(*routed.qregs, *routed.cregs)
269        for inst in routed.data:
270            if inst.operation.name != "dummy_cx":
271                cleaned_routed.append(inst)
272        return cleaned_routed

274    def _extract_sparse_swaps(self, global_circuit: QuantumCircuit, routed_sub_circuits: List[QuantumCircuit]) -> List[List[int]]:
275        sparse_swaps = []
276        for circ in routed_sub_circuits:
277            current_layer = 0
278            for inst in circ.data:
279                current_layer = self._update_current_layer(inst.operation, current_layer)
280                swap = self._get_sabre_swap_if_present(global_circuit, inst, current_layer)
281                if swap is not None:
282                    sparse_swaps.append(swap)
283        return sparse_swaps
284
285    def _update_current_layer(self, op: Instruction, current_layer: int) -> int:
286        if isinstance(op, (TeledataSend, TeledataRecv, TelegateHalf)):
287            return op.layer_idx
288            
289        if hasattr(op, 'label') and op.label and 'L' in op.label:
290            try:
291                return int(op.label.split('L')[-1])
292            except ValueError:
293                pass
294                
295        return current_layer

297    def _get_sabre_swap_if_present(self, global_circuit: QuantumCircuit, inst, current_layer: int) -> List[int] | None:
298        op = inst.operation
299        if op.name == 'swap' and not (hasattr(op, 'label') and op.label and 'L' in op.label):
300            # After ApplyLayout the circuit is physical: qubit indices are physical qubit indices.
301            # global_circuit.find_bit() cannot resolve them (different register), so read the
302            # index directly from the Qubit object instead.
303            q1 = inst.qubits[0]._index if hasattr(inst.qubits[0], '_index') else inst.qubits[0].index
304            q2 = inst.qubits[1]._index if hasattr(inst.qubits[1], '_index') else inst.qubits[1].index
305            return [current_layer, q1, q2]
306        return None
