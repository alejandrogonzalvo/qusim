 72    def get_qubit_fidelity_over_time(self, qubit: int) -> tuple[np.ndarray, np.ndarray, np.ndarray]:
 73        """
 74        Extracts the (algorithmic, routing, coherence) fidelity curves for a specific virtual qubit
 75        across all timeslices to graph hardware impacts dynamically.
 76        """
 77        algo_curve = self.algorithmic_fidelity_grid[:, qubit]
 78        route_curve = self.routing_fidelity_grid[:, qubit]
 79        coh_curve = self.coherence_fidelity_grid[:, qubit]
 80        return algo_curve, route_curve, coh_curve

 82    def get_circuit_fidelity_over_time(self) -> tuple[np.ndarray, np.ndarray, np.ndarray]:
 83        """
 84        Computes circuit-level fidelity curves by taking the product across all qubits at each
 85        layer. Returns (algorithmic, routing, coherence) arrays of shape (num_layers,).
 86        """
 87        algo_curve = np.prod(self.algorithmic_fidelity_grid, axis=1)
 88        route_curve = np.prod(self.routing_fidelity_grid, axis=1)
 89        coh_curve = np.prod(self.coherence_fidelity_grid, axis=1)
 90        return algo_curve, route_curve, coh_curve

210def _qiskit_circ_to_sparse_list(circ: qiskit.QuantumCircuit) -> tuple[np.ndarray, list[str]]:
211    """
212    Parses a Qiskit circuit's DAG into layers, and extracts interaction edges.
213    Returns a flat (E, 5) numpy array of [layer, q1, q2, weight, gate_type] and
214    a list of unique gate names indexed by gate_type.
215
216    Measure and barrier are excluded. Virtual gates (rz, id, delay) are preserved
217    to maintain graph layer alignment, but receive 0.0 duration and 0.0 error later.
218    """
219    dag = circuit_to_dag(circ)
220    layers = [dag_to_circuit(layer['graph']) for layer in dag.layers()]
221
222    edges = []
223    gate_names = []
224    gate_name_to_id = {}
225
226    for layer_idx, layer_as_circuit in enumerate(layers):
227        for instruction in layer_as_circuit:
228            name = instruction.operation.name
229            if name in {'measure', 'barrier'}:
230                continue
231            
232            if name not in gate_name_to_id:
233                gate_name_to_id[name] = len(gate_names)
234                gate_names.append(name)
235            gate_id = float(gate_name_to_id[name])
236
237            if instruction.operation.num_qubits == 2:
238                q0 = circ.find_bit(instruction.qubits[0]).index
239                q1 = circ.find_bit(instruction.qubits[1]).index
240                edges.append([float(layer_idx), float(q0), float(q1), 1.0, gate_id])
241            elif instruction.operation.num_qubits == 1:
242                q0 = circ.find_bit(instruction.qubits[0]).index
243                edges.append([float(layer_idx), float(q0), float(q0), 1.0, gate_id])
244
245    if len(edges) == 0:
246        return np.empty((0, 5), dtype=np.float64), gate_names
247
248    return np.array(edges, dtype=np.float64), gate_names

258def map_circuit(
259    circuit: qiskit.QuantumCircuit,
260    full_coupling_map: qiskit.transpiler.CouplingMap,
261    core_mapping: dict[int, int],
262    seed: Optional[int] = None,
263    initial_placement: InitialPlacement = InitialPlacement.RANDOM,
264    # Hardware defaults
265    single_gate_error: float = 1e-4,
266    two_gate_error: float = 1e-3,
267    teleportation_error_per_hop: float = 1e-2,
268    single_gate_time: float = 20.0,
269    two_gate_time: float = 100.0,
270    teleportation_time_per_hop: float = 1000.0,
271    epr_error_per_hop: float = 0.0,
272    measurement_error: float = 1e-3,
273    t1: float = 100_000.0,
274    t2: float = 50_000.0,
275    single_gate_error_per_qubit: Optional[np.ndarray] = None,
276    two_gate_error_per_pair: Optional[dict[tuple[int, int], float]] = None,
277    t1_per_qubit: Optional[np.ndarray] = None,
278    t2_per_qubit: Optional[np.ndarray] = None,
279    gate_error_per_type: Optional[dict[str, float]] = None,
280    gate_time_per_type: Optional[dict[str, float]] = None,
281    dynamic_decoupling: bool = False,
282    readout_error_per_qubit: Optional[np.ndarray] = None,
283    readout_mitigation_factor: float = 0.0,
284    classical_link_width: int = 0,
285    classical_clock_freq_hz: float = 200e6,
286    classical_routing_cycles: int = 2,
287) -> QuadrisResult:
288    """
289    Map a Qiskit quantum circuit using HQA and estimate teleportation routing and hardware fidelity.
290
291    This function calls the underlying compiled Rust engine for execution. The circuit is serialized
292    into a sparse interaction representation native to Numpy and fed across the ABI zero-copy layer
293    for nanosecond scalability profiling.
294
295    Args:
296        circuit (qiskit.QuantumCircuit): Qiskit QuantumCircuit comprising of your algorithms gate-suite.
297        full_coupling_map (qiskit.transpiler.CouplingMap): Hardware graph of the entire multi-core system.
298        core_mapping (dict[int, int]): Dictionary assigning physical qubit indices to core IDs.
299        seed (Optional[int]): If provided, ensures deterministic random seeding for the HQA initial state partition.
300        initial_placement (InitialPlacement): Policy for generating the layer 0 assignment heuristic.
301        
302        single_gate_error (float): 1Q physical fidelity loss rate.
303        two_gate_error (float): 2Q local operational fidelity loss rate.
304        teleportation_error_per_hop (float): Inter-core spatial fidelity degradation factor crossing distance parameters.
305        single_gate_time (float): Nanosecond processing overhead per 1Q operation.
306        two_gate_time (float): Nanosecond latency duration for processing local entanglements.
307        teleportation_time_per_hop (float): Communication EPR networking traversal nanoseconds dictating `T1`/`T2` accumulation scaling rates.
308        t1 (float): Hardware T1 characteristic exponential scaling time constant.
309        t2 (float): Hardware T2 decoherence characteristic timeline.
310        single_gate_error_per_qubit (Optional[np.ndarray]): Per-qubit 1Q error rates, shape (num_qubits,). Falls back to scalar single_gate_error if None.
311        two_gate_error_per_pair (Optional[dict[tuple[int,int], float]]): Per-pair 2Q error rates keyed by (q1, q2). Falls back to scalar two_gate_error for missing pairs.
312        t1_per_qubit (Optional[np.ndarray]): Per-qubit T1 relaxation times in ns.
313        t2_per_qubit (Optional[np.ndarray]): Per-qubit T2 dephasing times in ns.
314
315    Returns:
316        QuadrisResult: Structured simulation payload metrics including multidimensional matrices isolating individual qubit drop footprints across layers.
317    """
318    # 1. Parse Circuit to NumPy slices
319    gs_sparse, gate_names = _qiskit_circ_to_sparse_list(circuit)
320
321    _default_errors = {'rz': 0.0, 'id': 0.0, 'delay': 0.0}
322    _default_times = {'rz': 0.0, 'id': 0.0, 'delay': 0.0}
323    merged_errors = {**_default_errors, **(gate_error_per_type or {})}
324    merged_times = {**_default_times, **(gate_time_per_type or {})}
325
326    gate_error_arr = np.array([merged_errors.get(name, np.nan) for name in gate_names], dtype=np.float64)
327    gate_time_arr = np.array([merged_times.get(name, np.nan) for name in gate_names], dtype=np.float64)
328
329    # 2. Setup inputs
330    num_cores = max(core_mapping.values()) + 1 if core_mapping else 0
331    core_caps = np.zeros(num_cores, dtype=np.uint64)
332    for q, c in core_mapping.items():
333        core_caps[c] += 1
334        
335    # Derive distance matrix from full_coupling_map
336    core_adj = np.zeros((num_cores, num_cores), dtype=np.int32)
337    for edge in full_coupling_map.get_edges():
338        c1 = core_mapping[edge[0]]
339        c2 = core_mapping[edge[1]]
340        if c1 != c2:
341            core_adj[c1, c2] = 1
342            core_adj[c2, c1] = 1
343            
344    dist_mat = np.full((num_cores, num_cores), fill_value=9999, dtype=np.int32)
345    for c in range(num_cores):
346        dist_mat[c, c] = 0
347    for c1 in range(num_cores):
348        for c2 in range(num_cores):
349            if core_adj[c1, c2]:
350                dist_mat[c1, c2] = 1
351    
352    for k in range(num_cores):
353        for i in range(num_cores):
354            for j in range(num_cores):
355                if dist_mat[i, k] + dist_mat[k, j] < dist_mat[i, j]:
356                    dist_mat[i, j] = dist_mat[i, k] + dist_mat[k, j]
357        
358    num_virtual_qubits = circuit.num_qubits
359        
360    config = PlacementConfig(
361        policy=initial_placement,
362        interaction_tensor=gs_sparse,
363        num_virtual_qubits=num_virtual_qubits,
364        core_caps=core_caps,
365        seed=seed
366    )
367    initial_partition = generate_initial_placement(config)
368
369    # 3. Call Rust Engine for HQA pass and base estimations
370    raw_dict = map_and_estimate(
371        gs_sparse=gs_sparse,
372        initial_partition=initial_partition,
373        num_cores=num_cores,
374        core_capacities=core_caps,
375        distance_matrix=dist_mat,
376        single_gate_error=single_gate_error,
377        two_gate_error=two_gate_error,
378        teleportation_error_per_hop=teleportation_error_per_hop,
379        single_gate_time=single_gate_time,
380        two_gate_time=two_gate_time,
381        teleportation_time_per_hop=teleportation_time_per_hop,
382        epr_error_per_hop=epr_error_per_hop,
383        measurement_error=measurement_error,
384        t1=t1,
385        t2=t2,
386        single_gate_error_per_qubit=single_gate_error_per_qubit,
387        two_gate_error_per_pair=two_gate_error_per_pair,
388        t1_per_qubit=t1_per_qubit,
389        t2_per_qubit=t2_per_qubit,
390        gate_error_per_type=gate_error_arr,
391        gate_time_per_type=gate_time_arr,
392        dynamic_decoupling=dynamic_decoupling,
393        readout_error_per_qubit=readout_error_per_qubit,
394        readout_mitigation_factor=readout_mitigation_factor,
395        classical_link_width=classical_link_width,
396        classical_clock_freq_hz=classical_clock_freq_hz,
397        classical_routing_cycles=classical_routing_cycles,
398    )
399
400    placements = raw_dict["placements"]
401    num_layers = placements.shape[0] - 1
402    sparse_swaps_list = []
403
404    # 4. Sabre Orchestration Pass
405    if full_coupling_map is not None:
406        
407        # HQA list is indexed [timeslice][qubit] -> core_id
408        hqa_list = placements.tolist()
409        orchestrator = MultiCoreOrchestrator(num_cores, full_coupling_map, core_mapping, dist_mat)
410        
411        # Run orchestrator and grab the exact logical timestamps of where SABRE injected SWAPs!
412        _, sparse_swaps_list = orchestrator.orchestrate(circuit, hqa_list)
413
414    # 5. Fast-Path Fidelity Re-Estimation (incorporates newly resolved SWAPs natively)
415    if not sparse_swaps_list:
416        sparse_swaps_arr = np.zeros((0, 3), dtype=np.int32)
417    else:
418        sparse_swaps_arr = np.array(sparse_swaps_list, dtype=np.int32)
419
420    fidelity_dict = estimate_hardware_fidelity(
421        gs_sparse=gs_sparse,
422        placements=placements,
423        distance_matrix=dist_mat,
424        sparse_swaps=sparse_swaps_arr,
425        single_gate_error=single_gate_error,
426        two_gate_error=two_gate_error,
427        teleportation_error_per_hop=teleportation_error_per_hop,
428        single_gate_time=single_gate_time,
429        two_gate_time=two_gate_time,
430        teleportation_time_per_hop=teleportation_time_per_hop,
431        epr_error_per_hop=epr_error_per_hop,
432        measurement_error=measurement_error,
433        t1=t1,
434        t2=t2,
435        single_gate_error_per_qubit=single_gate_error_per_qubit,
436        two_gate_error_per_pair=two_gate_error_per_pair,
437        t1_per_qubit=t1_per_qubit,
438        t2_per_qubit=t2_per_qubit,
439        gate_error_per_type=gate_error_arr,
440        gate_time_per_type=gate_time_arr,
441        dynamic_decoupling=dynamic_decoupling,
442        readout_error_per_qubit=readout_error_per_qubit,
443        readout_mitigation_factor=readout_mitigation_factor,
444        classical_link_width=classical_link_width,
445        classical_clock_freq_hz=classical_clock_freq_hz,
446        classical_routing_cycles=classical_routing_cycles,
447    )
448
449    # 6. Wrap Results
450    return QuadrisResult(
451        execution_success=raw_dict.get("execution_success", False),
452        placements=placements,
453        total_teleportations=raw_dict["total_teleportations"],
454        total_swaps=sparse_swaps_arr.shape[0],
455        total_epr_pairs=raw_dict["total_epr_pairs"],
456        total_network_distance=raw_dict["total_network_distance"],
457        teleportations_per_slice=raw_dict["teleportations_per_slice"],
458
459        algorithmic_fidelity=fidelity_dict["algorithmic_fidelity"],
460        routing_fidelity=fidelity_dict["routing_fidelity"],
461        coherence_fidelity=fidelity_dict["coherence_fidelity"],
462        overall_fidelity=fidelity_dict["overall_fidelity"],
463        total_circuit_time_ns=fidelity_dict["total_circuit_time_ns"],
464
465        algorithmic_fidelity_grid=fidelity_dict["algorithmic_fidelity_grid"],
466        routing_fidelity_grid=fidelity_dict["routing_fidelity_grid"],
467        coherence_fidelity_grid=fidelity_dict["coherence_fidelity_grid"],
468
